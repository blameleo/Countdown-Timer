let displayTimeLeft = document.querySelector(".display__time-left"); // Display time 
// getting buttons !
let buttons = document.querySelectorAll(".timer__button");
let displayEndTime = document.querySelector(".display__end-time");
// console.log(buttons); // consoling node lists of buttons

const startingTime = 0.34; // getting 20 seconds as starting time
let time = startingTime * 60; // getting my seconds in millseconds


function countDownTime20sec() {
    const countDownTimer = setInterval(() => {
        let minutes = Math.floor(time / 60);
        let seconds = Math.floor(time % 60);
        // console.log(seconds);
        minutes = minutes < 10 ? "0" + minutes : minutes;
        seconds = seconds < 10 ? "0" + seconds : seconds;
        displayTimeLeft.textContent = `${minutes}: ${seconds}`;
        displayEndTime.textContent = `Be back at 00:00`
        time--
        if (seconds <= 0) {
            clearInterval(countDownTimer);
            time = startingTime * 60;
        }
    }, 1000);
};
